buy
######################

affordable = floor(count / cost)

price check
{
	WHILE target_amount > 0

		FOREACH in cost

			IF affordable >= target_amount
				next

			ELSEIF affordable >= 1
				actual amount = affordable
				next

			ELSE
				target_amount = 0
}

purchase
{
	IF target_amount > 0

		FOREACH in cost
			subtract price * target_amount
}

return target_amount
